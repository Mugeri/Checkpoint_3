/*
 * generate a token once a user logs in

function generateToken(userName, role) {
  const claims = {
    sub: userName,
    iss: 'docman',
    permissions: role,
  };
  const jwt = nJwt.create(claims, process.env.SECRET);
  const token = jwt.compact();
  return token;
}
